import { Injectable, UnauthorizedException } from '@nestjs/common';
import { PassportStrategy } from '@nestjs/passport';
import { Strategy } from 'passport-http-bearer'; // Asegúrate de instalar: passport-http-bearer
import * as admin from 'firebase-admin';
import { AuthService } from '../auth.service';

@Injectable()
export class FirebaseStrategy extends PassportStrategy(Strategy, 'firebase') {
  constructor(private authService: AuthService) {
    super();
  }

  async validate(token: string) {
    try {
      // Valida el token contra Google
      const decodedToken = await admin.auth().verifyIdToken(token);
      // Sincroniza con MySQL usando tu servicio de auth
      return await this.authService.syncUserWithFirebase(decodedToken);
    } catch (error) {
      throw new UnauthorizedException('Token de Firebase inválido');
    }
  }
}

1. Refactorización del AuthService (Lógica de Auto-Rol)
Ubicación: apps/api-gateway/src/app/modules/core-engine/auth/auth.service.ts
typescript
import { Injectable, InternalServerErrorException, Logger } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { UserEntity, RoleEntity } from '@arcani/data-access/db-entities';

@Injectable()
export class AuthService {
  private readonly logger = new Logger(AuthService.name);

  constructor(
    @InjectRepository(UserEntity)
    private readonly userRepo: Repository<UserEntity>,
    @InjectRepository(RoleEntity)
    private readonly roleRepo: Repository<RoleEntity>,
  ) {}

  /**
   * Sincroniza la identidad y asigna el rol de CUSTOMER por defecto.
   */
  async syncUserWithFirebase(fbUser: any): Promise<UserEntity> {
    try {
      // 1. Buscamos al usuario en MySQL
      let user = await this.userRepo.findOne({ 
        where: { firebaseUid: fbUser.uid },
        relations: ['role', 'role.level'] // Traemos toda la jerarquía de permisos
      });

      if (!user) {
        this.logger.log(`[AUTH] Sintonizando nuevo cliente: ${fbUser.email}`);

        // 2. BUSQUEDA DINÁMICA DEL ROL CUSTOMER
        // Buscamos en la tabla de roles aquel cuyo nivel asociado sea 'CUSTOMER'
        const customerRole = await this.roleRepo.findOne({
          where: { level: { name: 'CUSTOMER' } }
        });

        if (!customerRole) {
          this.logger.error('[CRÍTICO] No se encontró el nivel CUSTOMER en la base de datos.');
          throw new InternalServerErrorException('Falla de Configuración: El nivel CUSTOMER no existe.');
        }

        // 3. CREACIÓN CON ROL AUTOMÁTICO
        user = this.userRepo.create({
          firebaseUid: fbUser.uid,
          email: fbUser.email,
          name: fbUser.name || fbUser.displayName || 'Explorador',
          photoURL: fbUser.picture || fbUser.photoURL || null,
          idRole: customerRole.id, // Asignamos el ID del rol que encontramos
          isActive: true
        });
      } else {
        // Actualizamos datos básicos si el usuario ya existía
        user.name = fbUser.name || fbUser.displayName || user.name;
        user.photoURL = fbUser.picture || fbUser.photoURL || user.photoURL;
      }

      return await this.userRepo.save(user);

    } catch (error) {
      this.logger.error(`[AUTH ERROR] ${error.message}`);
      throw new InternalServerErrorException('Error al sincronizar identidad con el Oráculo.');
    }
  }
}

3. Consideraciones de Base de Datos (Seeds)
Para que este código no lance el error de "No se encontró el nivel CUSTOMER", debes asegurarte de que tu base de datos tenga estos datos sembrados (Seeds):
Tabla role_level: Debe existir una fila con name: 'CUSTOMER'.
Tabla role: Debe existir una fila (ej. name: 'Explorador') asociada al id_level del registro anterior.
💡 ¿Por qué es mejor este enfoque?
Desacoplamiento: No hardcodeamos un ID (como idRole: 1). Si en el futuro cambias los IDs en la base de datos, el código seguirá funcionando porque busca por el nombre del nivel (CUSTOMER).
Escalabilidad: El administrador podrá crear roles como "VIP" o "Socio" bajo el mismo nivel de "CUSTOMER" y la lógica de creación seguirá siendo sólida.


on esta misma logica, el admin puede otorgar permisos a un trabajador de la empresa, primero se loguea como cliente para saber sus datos por firebase correo y esos, luego el admin le asigna un rol, por ejemplo cajero. verdad?
Exactamente. Esa es la lógica RBAC (Control de Acceso Basado en Roles) profesional que estamos construyendo. El flujo sería el siguiente:
Registro Inicial: El trabajador entra al portal y se loguea con Google (Firebase). El AuthService lo detecta como nuevo y, por defecto, le asigna el nivel CUSTOMER.
Identificación: En ese momento, ya tienes su email y su uuid en tu base de datos MySQL.
Elevación de Rango: El administrador (desde su Dashboard) busca al usuario por correo y cambia su id_role al de Cajero (que pertenecerá a un nivel OPERATIVE o TACTICAL).
Actualización: La próxima vez que el trabajador se loguee, el AuthService encontrará que ya existe y cargará su nuevo rol con todos sus permisos.

import { ExecutionContext, Injectable } from '@nestjs/common';
import { Reflector } from '@nestjs/core';
import { AuthGuard } from '@nestjs/passport';
import { IS_PUBLIC_KEY } from '../decorators/public.decorator';

@Injectable()
export class FirebaseAuthGuard extends AuthGuard('firebase') {
  constructor(private reflector: Reflector) {
    super();
  }

  canActivate(context: ExecutionContext) {
    // 1. Verificamos si la ruta o el controlador tienen el decorador @Public()
    const isPublic = this.reflector.getAllAndOverride<boolean>(IS_PUBLIC_KEY, [
      context.getHandler(),
      context.getClass(),
    ]);

    // 2. Si es pública, permitimos el acceso sin validar token
    if (isPublic) {
      return true;
    }

    // 3. Si NO es pública, ejecutamos la validación de Firebase (Bearer Token)
    return super.canActivate(context);
  }
}

Para que el Oráculo reciba el token de identidad, necesitamos configurar el Interceptor de Angular. Este interceptor atrapará el token de Firebase y lo inyectará en la cabecera Authorization: Bearer <token> de cada petición HTTP.
Aquí tienes los pasos para conectar el Frontend con el Escudo que acabamos de crear en el Backend:
1. El Interceptor de Identidad (Angular)
Este archivo se encarga de adjuntar tu "pasaporte" digital en cada llamada a la API.
Ubicación: libs/shared/auth/src/lib/interceptors/auth.interceptor.ts
typescript
import { HttpInterceptorFn } from '@angular/common/http';
import { inject } from '@angular/core';
import { AuthService } from '../auth.service';
import { switchMap, take } from 'rxjs';

/**
 * INTERCEPTOR DE IDENTIDAD (FRONTEND)
 * Recupera el token de Firebase y lo inyecta como Bearer Token
 * para que el Oráculo reconozca al explorador.
 */
export const authInterceptor: HttpInterceptorFn = (req, next) => {
  const authService = inject(AuthService);

  // 1. Obtenemos el token del Signal/Observable de Firebase
  // Usamos pipe para esperar a que el token esté listo
  return authService.token$.pipe(
    take(1),
    switchMap((token) => {
      // 2. Si hay token, clonamos la petición con la cabecera de seguridad
      if (token) {
        const authReq = req.clone({
          setHeaders: {
            Authorization: `Bearer ${token}`
          }
        });
        return next(authReq);
      }

      // 3. Si no hay token (usuario anónimo), la petición sigue su curso normal
      return next(req);
    })
  );
};

Para cerrar el círculo de identidad, crearemos la interfaz UserIdentity en tu librería de dominio. Esta interfaz es el "espejo" exacto de lo que el Oráculo envía, permitiendo que Angular reconozca los permisos ofuscados y el rol del usuario con total seguridad de tipos.
1. Crear la Interfaz de Identidad
Ubicación: libs/shared/domain/src/lib/interfaces/user-identity.interface.ts
typescript
/**
 * IDENTIDAD DEL EXPLORADOR (ARCANI)
 * Esta interfaz mapea la respuesta blindada del Oráculo.
 */
export interface UserIdentity {
  uuid: string;           // Identificador público seguro (UUID)
  email: string;          // Correo del explorador
  name: string;           // Nombre sintonizado desde Google/MySQL
  photoURL: string | null;// Avatar del explorador
  roleName: string;       // Nombre del rol: "Director", "Cliente", etc.
  
  /**
   * CAPACIDADES (Permissions)
   * Lista de etiquetas (algunas ofuscadas) que habilitan 
   * funcionalidades en la interfaz.
   */
  permissions: string[]; 
}
Usa el código con precaución.
2. Actualizar el Punto de Exportación
Para que el AuthService y los componentes puedan ver esta interfaz, debemos exportarla en el índice de la librería.
Ubicación: libs/shared/domain/src/index.ts
typescript
// ... otras exportaciones
export * from './lib/interfaces/user-identity.interface';
